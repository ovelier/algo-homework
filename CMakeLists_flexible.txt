cmake_minimum_required(VERSION 3.19)
project(lab-scheduler-test LANGUAGES C CXX)

# 添加 SQLite 库
add_library(sqlite3 STATIC
    third_party/sqlite/sqlite3.c
)

target_include_directories(sqlite3 PUBLIC
    third_party/sqlite
)

# 测试程序(不需要Qt)
add_executable(test_algorithm
    src/test_algorithm.cpp
    src/database.cpp
    src/scheduler.cpp
)

target_include_directories(test_algorithm PRIVATE
    src
    third_party/sqlite
)

target_link_libraries(test_algorithm
    PRIVATE
        sqlite3
)

# 主程序(需要Qt)
find_package(Qt6 6.5 QUIET COMPONENTS Core Widgets)

if(Qt6_FOUND)
    qt_standard_project_setup()
    
    qt_add_executable(algo-homework
        WIN32 MACOSX_BUNDLE
        src/main.cpp
        src/widget.cpp
        src/widget.h
        src/database.cpp
        src/database.h
        src/scheduler.cpp
        src/scheduler.h
    )
    
    target_link_libraries(algo-homework
        PRIVATE
            Qt::Core
            Qt::Widgets
            sqlite3
    )
    
    include(GNUInstallDirs)
    
    install(TARGETS algo-homework
        BUNDLE  DESTINATION .
        RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
        LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
    )
    
    qt_generate_deploy_app_script(
        TARGET algo-homework
        OUTPUT_SCRIPT deploy_script
        NO_UNSUPPORTED_PLATFORM_ERROR
    )
    install(SCRIPT ${deploy_script})
else()
    message(WARNING "Qt6 not found. Skipping GUI application build. Only test_algorithm will be built.")
endif()
